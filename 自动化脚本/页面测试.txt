#!/bin/bash
#############################################################################################
#建连时间: %{time_connect} = 建连时间-dns解析时间                                           #
#首包相应：%{time_starttransfer} = 首包 - 建连                                              #
# time_starttransfer 开始传输时间。在发出请求之后，Web 服务器返回数据的第一个字节所用的时间 #
#%{time_pretransfer} 从client发出请求；到web的server 响应第一个字节的时间                   #
#页面大小: %{size_download}    单位：byte (B)
#时间 都是ms
#下载速度  单位-字节每秒。(Bytes per second)
#############################################################################################
#curl -o /dev/null -s -w "\
#网站状态: %{http_code}
#解析用时: %{time_namelookup}
#建连用时: %{time_connect}
#首包用时：%{time_starttransfer}
#页面耗时: %{time_total}
#页面大小: %{size_download}
#下载速度: %{speed_download}\n" http://www.baidu.com

name1=`curl -o /dev/null -s -w "%{http_code} %{time_namelookup} %{time_connect} %{time_starttransfer} %{time_total} %{size_download} %{speed_download}\n" http://www.baidu.com`
name2=($name1)
echo "网站状态: ${name2[0]}
解析用时: ${name2[1]}
建连用时: ${name2[2]}
首包用时: ${name2[3]}
页面耗时: ${name2[4]}
页面大小: ${name2[5]}
下载速度: ${name2[6]}"
